class User(models.Model):
	username	= models.CharField(max_length = 32)

	fname		= models.CharField(max_length = 32)
	lname		= models.CharField(max_length = 32)
	
	gender		= models.CharField(max_length = 1, blank = True)
	dob			= models.DateField("Date of Birth", blank = True)
	zip			= models.IntegerField(max_length = 5, blank = True)
	email		= models.EmailField(max_length = 254, blank = True)
	address		= models.TextField(blank = True)
	rating		= models.DecimalField(max_digits = 3, decimal_places = 2, blank = True)
	type		= models.IntegerField(max_length = 1, blank = True)
	
	zone		= models.ForeignKey(ShareZone, related_name = "ShareZone", blank = True)
	
	def __str__(self):
		return self.name
